Gramatica
parametro c:Real valor 1.0
parametro p:Real valor 0.3
parametro q:Real := c-p
parametro h:Real := (p*q)^0.2
simbolo F(x,t)

pasos 10
angulo 86
posicion (0,0,0,0) 90

matriz
#F(1,0)
